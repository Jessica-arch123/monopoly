function handleLanding(p) { const tileId = p.pos; if(tileId === 0) { p.pts += 100; play('correct'); endTurn(); return; } // Check if Tile is an AUDIT (Penalty) tile if(tileNames[tileId] === "AUDIT") { p.pts -= 40; alert("🚨 Safety Violation! -40 PTS"); endTurn(); return; } // Check if Tile is OWNED by someone else if(owners[tileId] !== undefined && owners[tileId] !== turn){ let owner = players[owners[tileId]]; let rent = 60; // 1. Transfer Point to Owner p.pts -= rent; owner.pts += rent; alert(`🏠 LANDED ON OWNED LAND! Paid ${rent} PTS to ${owner.icon}`); // 2. Structural Penalty Question setTimeout(() => { alert(`🧪 STRUCTURAL AUDIT: Answer correctly or lose 30 extra PTS!`); showQuestion(tileId, p, true); // Added 'true' for penalty mode }, 800); } else { // Tile is unowned or owned by the player themselves showQuestion(tileId, p, false); } } function showQuestion(tid, p, isPenalty){ // Filter questions (You can filter by lvl if you want specifically Structural) const q = questions[Math.floor(Math.random()*questions.length)]; document.getElementById('qOverlay').style.display = 'flex'; document.getElementById('qHeader').innerText = isPenalty ? "STRUCTURAL PENALTY" : "RESEARCH TASK"; document.getElementById('qText').innerHTML = q.text; const area = document.getElementById('optionsArea'); area.innerHTML = ''; q.options.forEach(opt => { const b = document.createElement('button'); b.className = 'btn-mcq'; b.innerHTML = opt; b.onclick = () => { if(opt === q.correct){ play('correct'); if(!isPenalty && owners[tid] === undefined){ owners[tid] = turn; document.getElementById('t'+tid).innerHTML += `${p.icon}`; } else if(!isPenalty) { p.pts += 30; // Reward for correct answer on own/free land } alert("Correct! Task Successful."); } else { play('wrong'); if(isPenalty){ p.pts -= 30; // Additional penalty for failing the audit alert("FAILED AUDIT: Lost 30 extra PTS."); } } document.getElementById('qOverlay').style.display = 'none'; checkWin(); endTurn(); }; area.appendChild(b); }); }
